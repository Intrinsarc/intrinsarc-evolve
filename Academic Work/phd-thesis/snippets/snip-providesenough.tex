pred providesEnough(s: Stratum,
    provided: set Interface, required: set Interface)
{
  all prov: provided |
    one req: required |
      req in prov.*(superTypes.s)
  all req: required |
    one prov: provided |
      req in prov.*(superTypes.s)
}
